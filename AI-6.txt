def evaluate_employee():
    print("=== Employee Performance Evaluation System ===\n")

    name = input("Enter Employee Name: ")

   
    punctuality = int(input("Rate punctuality (1-10): "))
    teamwork = int(input("Rate teamwork (1-10): "))
    productivity = int(input("Rate productivity (1-10): "))
    communication = int(input("Rate communication skills (1-10): "))
    innovation = int(input("Rate innovation (1-10): "))

    score = (
        punctuality * 0.15 +
        teamwork * 0.20 +
        productivity * 0.30 +
        communication * 0.20 +
        innovation * 0.15
    )

  
    if score >= 8:
        performance = "Excellent"
    elif score >= 6:
        performance = "Good"
    elif score >= 4:
        performance = "Average"
    else:
        performance = "Poor"

   
    suggestions = []

    if productivity >= 9 and innovation >= 8:
        suggestions.append("Eligible for promotion or leadership role")

    if teamwork < 5:
        suggestions.append("Needs teamwork improvement training")

    if communication < 5:
        suggestions.append("Improve communication skills")

    if punctuality < 5:
        suggestions.append("Work on punctuality and discipline")

    if score < 5:
        suggestions.append("Overall performance improvement plan required")

    
    print("\n=== Evaluation Result ===")
    print("Employee Name:", name)
    print("Final Score:", round(score, 2))
    print("Performance Level:", performance)

    print("\nExpert Suggestions:")
    if suggestions:
        for s in suggestions:
            print("-", s)
    else:
        print("- Keep up the good work!")


evaluate_employee()
